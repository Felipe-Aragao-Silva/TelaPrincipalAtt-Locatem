// Exemplo de código C# no MainPage.xaml.cs
private void OnMenuItemTapped(object sender, EventArgs e)
{
    // Obtém o nome do item clicado
    var grid = sender as Grid;
    if (grid?.GestureRecognizers.FirstOrDefault() is TapGestureRecognizer tapGesture)
    {
        string menuItem = tapGesture.CommandParameter.ToString();
        
        // Aqui você faria a navegação ou trocaria a página principal
        Console.WriteLine($"Item clicado: {menuItem}");

        // Para a animação de seleção, você pode usar o VisualStateManager
        // VisualStateManager.GoToState(grid, "Selected"); 
    }
}
